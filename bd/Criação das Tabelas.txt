CREATE DATABASE bd_alugmat

CREATE TABLE PRODUTO (
ID INT PRIMARY KEY AUTO_INCREMENT,
DESCRICAO VARCHAR(50) NOT NULL,
CD_INTERNO VARCHAR(10) NOT NULL,
VALOR_DIARIA FLOAT NOT NULL,
STATUS CHAR(1) NOT NULL,
DISPONIVEL CHAR(1),
CARACTERISTICAS VARCHAR(65535),
MARCA VARCHAR(30) NOT NULL,
CATEGORIA VARCHAR(30) NOT NULL,
FORNECEDOR VARCHAR(50) NOT NULL,
DATA_INC DATETIME,
DATA_ALT DATETIME,
CD_CLIENTE INT,
NOTA FLOAT,
DESTAQUE CHAR(1)
)

CREATE TABLE CLIENTE (
ID INT PRIMARY KEY AUTO_INCREMENT,
TIPO_PESSOA CHAR(1) NOT NULL,
NOME VARCHAR(50) NOT NULL,
CPF VARCHAR(14),
RG VARCHAR(12),
CNPJ VARCHAR(18),
IE VARCHAR(15),
LOGRADOURO VARCHAR(20) NOT NULL,
NOME_LOGRADOURO VARCHAR(50) NOT NULL,
NUM INT NOT NULL,
COMPLEMENTO VARCHAR(20),
BAIRRO VARCHAR(30) NOT NULL,
CIDADE VARCHAR(40) NOT NULL,
ESTADO CHAR(2) NOT NULL,
TELEFONE VARCHAR(13),
CELULAR VARCHAR(14),
EMAIL VARCHAR(50) NOT NULL,
SENHA CHAR(40) NOT NULL,
DATA_INC DATETIME,
DATA_ALT DATETIME,
STATUS CHAR(1) NOT NULL,
TIPO_USUARIO CHAR(3)
)

CREATE TABLE COMENTARIOS (
ID INT PRIMARY KEY AUTO_INCREMENT,
CD_CLIENTE INT,
CD_PRODUTO INT NOT NULL,
NOTA INT NOT NULL,
COMENTARIO VARCHAR(65535) NOT NULL,
CONSTRAINT FK_CLIENTE FOREIGN KEY(CD_CLIENTE) REFERENCES CLIENTE(ID)
)
